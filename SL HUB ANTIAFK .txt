local function createStat(yPos, text)
	local label = Instance.new("TextLabel", Main)
	label.Size = UDim2.new(1,0,0,18)
	label.Position = UDim2.new(0,0,yPos,0)
	label.BackgroundTransparency = 1
	label.Font = Enum.Font.Gotham
	label.TextSize = 11
	label.TextColor3 = Color3.fromRGB(220,220,220)
	label.Text = text
	return label
end

local TimeLabel = createStat(0.22, "Time: 00:00:00")
local FPSLabel  = createStat(0.38, "FPS: 0")
local PingLabel = createStat(0.52, "Ping: 0 ms")

local BoostButton = Instance.new("TextButton", Main)
BoostButton.Size = UDim2.new(1,-16,0,22)
BoostButton.Position = UDim2.new(0,8,0.72,0)
BoostButton.BackgroundColor3 = Color3.fromRGB(25,25,25)
BoostButton.TextColor3 = Color3.fromRGB(255,255,255)
BoostButton.Font = Enum.Font.GothamBold
BoostButton.TextSize = 11
BoostButton.Text = "🚀 Boost: OFF"
Instance.new("UICorner", BoostButton)

local boosted = false

local function enableBoost()
	Lighting.GlobalShadows = false
	Lighting.FogEnd = 100000
	Lighting.Brightness = 1
	for _, v in pairs(workspace:GetDescendants()) do
		if v:IsA("BasePart") then
			v.Material = Enum.Material.Plastic
			v.Reflectance = 0
		end
		if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Smoke") or v:IsA("Fire") or v:IsA("Sparkles") then
			v.Enabled = false
		end
	end
	settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
end

BoostButton.MouseButton1Click:Connect(function()
	boosted = not boosted
	if boosted then
		enableBoost()
		BoostButton.Text = "🚀 Boost: ON"
	else
		BoostButton.Text = "🚀 Boost: OFF"
	end
end)

task.spawn(function()
	while true do
		task.wait(1)
		local elapsed = math.floor(tick() - startTime)
		local h = math.floor(elapsed/3600)
		local m = math.floor((elapsed%3600)/60)
		local s = elapsed%60
		TimeLabel.Text = string.format("⏳ TIME %02d:%02d:%02d",h,m,s)

		local fps = math.floor(1/RunService.RenderStepped:Wait())
		FPSLabel.Text = "🚀 FPS "..fps

		local ping = math.floor(Stats.Network.ServerStatsItem["Data Ping"]:GetValue())
		PingLabel.Text = "📶 PING "..ping.." ms"
	end
end)

	task.wait(1)
	vu:Button2Up(Vector2.new(0,0), workspace.CurrentCamera.CFrame)
end)
